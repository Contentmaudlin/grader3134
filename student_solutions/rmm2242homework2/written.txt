public void printLots(List<AnyType> L, List<Integer> P) {
	
	Iterator<Integer> itrP = P.iterator(); // create an iterator for P
	int nextP = itrP.next(); // get the first value of P using iterator
	int count = 0;
	for (int L_val:L) { // any out-of-bounds P indices just won't be reached
		if (nextP == count) { // if this L_val is at an index from P, it should be printed
			System.out.println(L_val);
			nextP = itrP.next(); // get the next P value 
		}
		count++; // use count to keep track of L index
	}
}
