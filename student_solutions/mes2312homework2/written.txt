private static <AnyType> void printLots(List<AnyType> L, List<Integer> P) {
	Iterator<AnyType> lit = L.iterator();
	Iterator<Integer> pit = P.iterator();
	
	int index = pit.next();
	int count = 0;
	
	while (lit.hasNext()) {
		if (count == index) {
			System.out.println(lit.next());
			
			if (pit.hasNext()) {
				index = pit.next();
				count++;
			}
			else {
				break;
			}
		}
		else {
			count++;
			lit.next();
		}
	}
}

private static List Intersect(L1,L2)
{
	Iterator L1it = L1.iterator();
	Iterator L2it = L2.iterator();
	List intersectList;
	AnyType lastValue = null;
	L1val = L1it.next();
	L2val = L2it.next();


	while (L1it.hasNext() OR L2it.hasNext())
	{
		if (L1val.compareTo(L2val) > 0)
		{
			L2val = L2it.next();
		}
		else if (L1val.compareTo(L2val) < 0)
		{
			L1val = L1it.next();
		}

		else
		{	
			//protects against duplicates
			if (lastValue == L1val) 
			{
				L2val = L2it.next();
				L1val = L1it.next();	
			}
			else
			{
				lastValue = L1val
				intersectList.add(L1val);
				if (L1it.hasNext() AND L2it.hasNext())
				{
					L1val = L1it.next()
					L2val = L2it.next()
				}
			}
		}
	}
	return intersectList;
}
