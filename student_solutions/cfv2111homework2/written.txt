This pseudocode implements two stacks using only one array. My stack routines declares an overflow when there is not enough space in the array 

public class TwoStacks {
	
	// Instance variables 
	private int leftpointer; 
	private int rightpointer; 
	private int size; 
	
	// Constructor 
	TwoStakcs(int length){
		// Create an empty generic array 
		<AnyType> twoStacks[] = new <AnyType> [length];
		
		// Set the size of the stack to length and 
		// have to pointers to point the beginning and end of the array 
		size = length; 
		leftPointer = -1; 
		rightPointer = size;
	}
	
	// method that checks if the array has enough space 
	private boolean enoughSpace (){
		
		return twoStacks.length > leftPointer + rightPointer; 
	}
	// Method that pushes elements to the right 
	pushRight (<AnyType> element) throws Exception{
		if (enoughSpace()){
			rightpointer --;
			twoStacks[rightPointer] = element; 
			
		}
		else { 
			throw new Exception();
		}
	}
	// Method that pushes elements to the left 
	pushLeft (<AnyType> element) throws Exception{
		if (enoughSpace()){
			leftPointer ++;
			twoStacks[leftPointer] = element; 
			
		}
		else { 
			throw new Exception();
		}
	}

	// Method that pops elements at the left 
	popLeft() throws Exception {
	    if (leftPointer < 0) {
	      throw new Exception();
	    } else {
	      return twoStacks[leftPointer];
	      leftPointer--;
	    }
	  }
	// Method that pops elements at the right 
	popRight() throws Exception {
	    if (rightPointer > size) {
	      throw new Exception();
	    } 
	    else {
	      return twoStacks[leftPointer];
	      rightPointer++;
	    }
	 }
}
