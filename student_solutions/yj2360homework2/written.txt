public class PrintLots {
	public static <T>  void printLots(List<T> l, List<Integer> p){
		Iterator<Integer> position = p.iterator();
		Iterator<T> element = l.iterator();
		int index = 0;
		while(position.hasNext()){
			try{
				int printIndex = position.next();
				for (int i = index; i < printIndex; i++) {
				
					element.next();
					index++;
				}
				System.out.println(element.next());
				index++;
			}
			catch(Exception e){return;}
		}
		
	}

		List<String> a = new ArrayList<String>();

	public static <T extends Comparable<T>> int search(List<T> arr, T x){
		int start;
		int stop;
		start = 0;
		stop = arr.size()-1;
		int mid;
		while(start<=stop){
			mid = (start+stop)/2;
			if (arr.get(mid).compareTo(x)>0) {
				stop=mid-1;
			}else if(arr.get(mid).compareTo(x)<0){
				start=mid+1;
			}else{
				return mid;
			}
		}
		return -1;
		
	}
