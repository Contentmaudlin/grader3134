public static void printLots(List<Integer> list1, List<Integer> list2) 
	{
		Object element = "Not in range";
		Iterator indexes = list2.iterator();
		Iterator list = list1.iterator();
		int index;
		int count = 0;
		while(indexes.hasNext())
		{
			index = (int) indexes.next();
			if(index == 0)
			{
				element = list.next();
				count++;
			}
			else if(index<list1.size() && index > 0)
			{
				for(int n=count; n<index+1; n++)
				{
					element = list.next();
					count++;
				}
			}
			else
			{
				element = "Not in range";
			}
			System.out.println(element);
		}
	}

{
	twoStacks(int n)
	{
		array = new Object array[n]
		size = n 
		top1=-1
		top2=n
	}

	pop1()
	{
		if(top1<0)
			return stack underflow error
		else
		{
			return array[top1]
			top1--
		}
	}

	pop2()
	{
		if (top2>=size)
			return stack underflow error
		else 
		{
			return array[top2]
			top2++
		}
	}
	push1(c)
	{
		if(top2-top1 == 1)
