public class twoStacksOneArray(){
	
	AnyType[] arr = (AnyType[]) new Object[10];

	int S2Lid = (arr.length - 1);

		if (((S2Lid - 1) - S1Lid) > 0){
			return true;
		} else {
			return false;
		}

	}

		}

		if (isSpace()){

		}
	}
